bool prim(ul x)
{
    if (x<2)
        return false;
    if (x>2 && x%2==0)
        return false;
    for(ui d=3; d*d<=x; d+=2)
        if(x%d==0)
            return false;
    return true;
}

void quicksort(ui left, ui right, ul q[])
{
    ui i=left, j=right, pivot;
    ul aux;
    pivot=q[(left+right)/2];
    while(i<j)
    {
        while(q[i]<pivot)
            i++;
        while(q[j]>pivot)
            j--;
        if(i<=j)
        {
            aux=q[i];
            q[i]=q[j];
            q[j]=aux;
            i++;
            j--;
        }
    }
    if(left<j)
        quicksort(left, j, q);
    if(i<right)
        quicksort(i, right, q);
}
